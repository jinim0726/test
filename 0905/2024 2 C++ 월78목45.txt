{

	// 3 x 1 = 3 쭉 ~~~

	std::cout << "구구단 중 몇 단을 출력하고 싶니?";

	int num;
	std::cin >> num;
	for (int i = 0; i < 9; i++)
	{
		std::cout << num << " X " << i + 1 << " = " << num * (i + 1) << std::endl;
	}

	save("main.cpp");
}
